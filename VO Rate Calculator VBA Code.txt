Sub RunProgam()
    Dim hs As Worksheet
    Set hs = ThisWorkbook.Sheets("HiddenSheet")
    ImportScript
    
    Dim CharName As String
    CharName = Trim(InputBox("Enter character name without spaces or colons:", "Character Extract"))
    If CharName = "" Then Exit Sub
    
    CharacterExtract hs, CharName
    
    Dim userChoice1 As String
    userChoice = InputBox("Choose what your rate is based on per line or per word. Type 1 or 2:" & vbCrLf & _
                          "1: Per line" & vbCrLf & _
                          "2: Per Word", "Selection Required")
    Dim RateType As String
    Select Case userChoice
        Case "1": RateType = "Line"
        Case "2": RateType = "Word"
        Case "": MsgBox "User cancelled"
        Case Else: MsgBox "Invalid input"
    End Select
    Dim Rate As Integer
    Rate = InputBox("Insert Rate per " & RateType & " without any monetary symbols.", "Rate Input", "0.00")
    
    BuildCharacterSummary hs, CharName, RateType, Rate
End Sub
Private Sub ImportScript()
    Dim FilePath As Variant
    Dim FileExt As String
    Dim FileContent As String
    Dim hs As Worksheet
    Dim Lines() As String
    Dim i As Long, Row As Long

    ' --- Allow both TXT and DOCX ---
    FilePath = Application.GetOpenFilename( _
        "Text & Word Files (*.txt;*.docx), *.txt;*.docx", , _
        "Select Script File (UTF-8 TXT or DOCX)")
    If FilePath = False Then Exit Sub

    FileExt = LCase(Mid(FilePath, InStrRev(FilePath, ".") + 1))

    ' --- Read file content ---
    Select Case FileExt
        Case "txt"
            FileContent = ReadTxtFile(CStr(FilePath))
        Case "docx"
            FileContent = ReadDocxFile(CStr(FilePath))
        Case Else
            MsgBox "Unsupported file type. Please select a .txt or .docx file.", vbExclamation
            Exit Sub
    End Select

    If FileContent = "" Then Exit Sub

    ' --- Get or prepare HiddenSheet ---
    On Error Resume Next
    Set hs = ThisWorkbook.Sheets("HiddenSheet")
    On Error GoTo 0
    If hs Is Nothing Then
        Set hs = ThisWorkbook.Sheets.Add
        hs.Name = "HiddenSheet"
    End If
    hs.Cells.Clear

    ' --- Normalize smart quotes (optional, keep from your original) ---
    'FileContent = Replace(FileContent, Chr(8220), """")  ' Left "
    'FileContent = Replace(FileContent, Chr(8221), """")  ' Right "

    ' --- Split by line ---
    Lines = SplitLines(FileContent)

    ' --- Dump lines into HiddenSheet ---
    Row = 1
    For i = LBound(Lines) To UBound(Lines)
        Dim lineVal As String
        lineVal = Trim(Lines(i))
        If lineVal <> "" Then
            hs.Cells(Row, 1).Value = lineVal
            Row = Row + 1
        End If
    Next i

    hs.Columns("A").WrapText = False
    MsgBox "Import complete with " & (Row - 1) & " lines imported.", vbInformation
End Sub

Private Function ReadDocxFile(FilePath As String) As String
    Dim wdApp As Object
    Dim wdDoc As Object
    Dim result As String
    
    On Error GoTo DocError
    
    ' Open Word invisibly
    Set wdApp = CreateObject("Word.Application")
    wdApp.Visible = False
    Set wdDoc = wdApp.Documents.Open(FilePath, ReadOnly:=True)
    
    ' Extract all text (preserves paragraph breaks as vbCr)
    result = wdDoc.content.text
    
    wdDoc.Close SaveChanges:=False
    wdApp.Quit
    Set wdDoc = Nothing
    Set wdApp = Nothing
    
    ReadDocxFile = result
    Exit Function
    
DocError:
    If Not wdDoc Is Nothing Then wdDoc.Close SaveChanges:=False
    If Not wdApp Is Nothing Then wdApp.Quit
    MsgBox "Error reading Word file: " & Err.Description, vbExclamation
    ReadDocxFile = ""
End Function

' Normalize all line ending variants to vbLf, then split
Private Function SplitLines(content As String) As String()
    Dim normalized As String
    normalized = Replace(content, vbCrLf, vbLf)   ' Windows CRLF ? LF
    normalized = Replace(normalized, vbCr, vbLf)   ' Word/Mac CR  ? LF
    SplitLines = Split(normalized, vbLf)
End Function

Private Function ReadTxtFile(FilePath As String) As String
    Dim FileNum As Integer, FileData As String
    FileNum = FreeFile
    Open FilePath For Input As #FileNum
    FileData = Input$(LOF(FileNum), FileNum)
    Close #FileNum
    ReadTxtFile = FileData
End Function

Sub CharacterExtract(hs As Worksheet, CharName As String)
    Dim i As Long
    Dim j As Long
    j = 1
    Dim lastRowA As Long
    lastRowA = hs.Cells(hs.Rows.Count, "A").End(xlUp).Row

    For i = 1 To lastRowA
        If LCase(Replace(hs.Cells(i, 1).Value, " ", "")) = LCase(Replace(CharName, " ", "")) & ":" Then
            hs.Cells(j, 2).Value = hs.Cells(i + 1, 1).Value
            j = j + 1
        End If
    Next i
End Sub

Sub BuildCharacterSummaryMulti(hs As Worksheet, CharName As String)

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets(1)

    Dim lastRowA As Long
    lastRowA = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    Dim lastRowC As Long
    lastRowC = ws.Cells(ws.Rows.Count, "C").End(xlUp).Row
    Dim LastRowB As Long
    LastRowB = hs.Cells(hs.Rows.Count, "B").End(xlUp).Row
    
     ' --- Clear old table --- '
    ws.Range("E1:K1").UnMerge
    ws.Range("D1:I1000").Clear
    ws.Range("K1:M1000").Clear
    Dim i As Long
    For i = 1 To lastRowC
        ws.Cells(i, 4).Value = "."
    Next i
    ' --- Headers --- '
    ws.Cells(1, 5).Value = "Character Summary"
    ws.Range("E2:K2").Value = Array("Character", "Lines", "Words", "Est. Time", "Priority", "Rate", "Est. Cost")
    
    ' --- Add characters to table --- '
    ws.Range("E3:E" & lastRowA + 2).Value = ws.Range("A1:A" & lastRowA).Value
    
    ' --- Add character data --- '
    Dim lastRowTable As Long, CharName As String, text As String, words As Long, Lines As Long
    lastRowTable = ws.Cells(ws.Rows.Count, "E").End(xlUp).Row
    words = 0
    Lines = 0
    For i = 3 To lastRowTable
        CharName = ws.Cells(i, 5).Value
        For j = 1 To lastRowC
            If ws.Cells(j, 2).Value = CharName Then
                Lines = Lines + 1
                text = ws.Cells(j, 3).Value
                words = words + CountWords(text)
            End If
        Next j
        ws.Cells(i, 6).Value = Lines
        ws.Cells(i, 7).Value = words
        ws.Cells(i, 8).Value = Round(words / 1.875) / 86400
        ws.Cells(i, 8).NumberFormat = "h:mm:ss"
        ws.Cells(i, 9).Formula = "=RANK(G" & i & ",G$3:G$" & lastRowTable & ",0)"
        ws.Cells(i, 11).Formula = "=IF(j" & i & "=0,0,IF(f" & i & "<10,10,f" & i & "*j" & i & "))"
        Lines = 0
        words = 0
    Next i

    ' --- Totals row --- '
    lastRowTable = ws.Cells(ws.Rows.Count, "E").End(xlUp).Row
    ws.Cells(lastRowTable + 1, 5).Value = "TOTALS"
    ws.Cells(lastRowTable + 1, 6).Formula = "=SUM(F3:F" & lastRowTable & ")"
    ws.Cells(lastRowTable + 1, 7).Formula = "=SUM(G3:G" & lastRowTable & ")"
    ws.Cells(lastRowTable + 1, 8).Formula = "=SUM(H3:H" & lastRowTable & ")"
    ws.Cells(lastRowTable + 1, 9).Value = "---"
    ws.Cells(lastRowTable + 1, 10).Value = "---"
    ws.Cells(lastRowTable + 1, 11).Formula = "=SUM(K3:K" & lastRowTable & ")"
    
    ' --- Formatting --- '
    ws.Range("E1:K1").Merge
    ws.Range("E1:K1").HorizontalAlignment = xlCenter
    ws.Range("E2:K2").Borders(xlEdgeBottom).LineStyle = xlContinuous
    ws.Range("E2:K2").Borders(xlEdgeBottom).Weight = xlMedium
    ws.Range("E" & lastRowTable & ":K" & lastRowTable).Borders(xlEdgeBottom).LineStyle = xlContinuous
    ws.Range("E" & lastRowTable & ":K" & lastRowTable).Borders(xlEdgeBottom).Weight = xlMedium
    ws.Range("J3:K" & lastRowTable + 1).NumberFormat = "$#0.00"
    ws.Columns("E:K").AutoFit
End Sub
Sub BuildCharacterSummary(hs As Worksheet, CharName As String, RateType As String, Rate As Integer)
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Output")
    ws.Cells.Clear

    Dim LastRowB As Long
    LastRowB = hs.Cells(hs.Rows.Count, "B").End(xlUp).Row

    ' --- Clear old table ---
    ws.Range("A1:G1000").Clear

    ' --- Headers ---
    ws.Cells(1, 1).Value = "Character Summary"
    ws.Range("A2:G2").Value = Array("Character", "Lines", "Words", "Est. Time", "Priority", "Rate per " & RateType, "Est. Cost")

    ' --- Single character row ---
    ws.Cells(3, 1).Value = CharName

    ' --- Count lines and words from HiddenSheet column B ---
    Dim i As Long
    Dim text As String
    Dim words As Long
    Dim Lines As Long
    words = 0
    Lines = 0

    For i = 1 To LastRowB
        text = hs.Cells(i, 2).Value
        If Trim(text) <> "" Then
            Lines = Lines + 1
            words = words + CountWords(text)
        End If
    Next i

    ' --- Write character data ---
    ws.Cells(3, 2).Value = Lines
    ws.Cells(3, 3).Value = words
    ws.Cells(3, 4).Value = Round(words / 1.875) / 86400
    ws.Cells(3, 4).NumberFormat = "h:mm:ss"
    ws.Cells(3, 5).Value = 1
    ws.Cells(3, 6).Value = Rate
    ws.Cells(3, 7).Formula = "=IF(F3=0,0,IF(B3<10,10,B3*F3))" ' Est. Cost

    ' --- Formatting ---
    ws.Range("A1:G1").Merge
    ws.Range("A1:G1").HorizontalAlignment = xlCenter
    ws.Range("A2:G2").Borders(xlEdgeBottom).LineStyle = xlContinuous
    ws.Range("A2:G2").Borders(xlEdgeBottom).Weight = xlMedium
    ws.Range("F3:G3").NumberFormat = "$#0.00"
    ws.Columns("A:G").AutoFit
End Sub
Private Function CountWords(ByVal txt As String) As Long
    txt = Application.WorksheetFunction.Trim(txt)
    CountWords = UBound(Split(txt, " ")) + 1
End Function
